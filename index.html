GANJIL GENAP

FOR

for (let i = 1 ; i <= 5; i++){
if (i % 2 != 0){
console.log (`${i} = bilangan ganjil`)
} else {
console.log (i, "= bilangan genap")
}
}

WHILE

let o = 1
while (o <= 5){
if (o % 2 != 0){
console.log (`${o} = bilangan ganjil`)
} else {
console.log (`${o} = bilangan genap`)
}
o++
}

DO WHILE

let p = 1
do {
if (p % 2 != 0){
console.log ('bilangan ganjil =', p)
} else {
console.log ('bilangan genap =', p)
}
p++
} while (p <= 5)

FAKTORIAL

FOR

let n = 5
let faktorial = 1
for(let i = 1; i <= n; i++){
faktorial *= i
console.log (faktorial)
}

WHILE

let a = 5
let faktor = 1
let f = 1
while ( f <= a ){
faktor *= f
console.log (faktor)
f++
}

Do While

let m = 5
let faktori = 1
let k = 1
do {
faktori *= k
console.log (faktori)
k++
}while (k <= m)
